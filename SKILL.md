---
name: opnsense-ingest
description: >-
  Ingest findings from opnDossier for OPNsense firewalls. Processes JSON firewall analysis
  and stores snapshots with metadata (host, timestamp, source) in a safe directory for later queries.
version: 1.0.0
triggers:
  - "ingest opnsense"
  - "store opnsense findings"
  - "opnDossier snapshot"
  - "firewall ingest"
  - "process OPNsense config"
permissions:
  - filesystem
  - read
  - write
inputs:
  - name: firewallHost
    type: string
    required: true
    description: Hostname or IP address of the OPNsense firewall (e.g. fw.node.org)
  - name: snapshotTime
    type: string
    required: true
    description: Snapshot timestamp in format YYYYMMDD-HHMMSS
  - name: source
    type: string
    required: true
    description: Analysis source (usually "opnDossier")
  - name: findings
    type: object
    required: true
    description: Full findings JSON generated by opnDossier convert -f json
outputs:
  - name: snapshotPath
    type: string
    description: Full path where the snapshot JSON file was stored
  - name: findingsSummary
    type: string
    description: High-level summary of critical findings (vulnerabilities, dead rules, etc.)
  - name: status
    type: string
    description: "ok" if the snapshot was saved successfully, "error" otherwise
---
